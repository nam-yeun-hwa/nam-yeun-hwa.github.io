рҹ“қ мӮ¬мҡ©мһҗ кІҪн—ҳ(UX) к°ңм„ 

мң„м—җм„ңлҸ„ мқҙм•јкё° н–Ҳм§Җл§Ң л№„лҸҷкё° мІҳлҰ¬лҘј нҶөн•ҙ мӢңк°„мқҙ кұёлҰ¬лҠ” мһ‘м—…мқ„ л°ұк·ёлқјмҡҙл“ңм—җм„ң мӢӨн–үн•ҳм—¬, к·ё лҸҷм•Ҳ мӮ¬мҡ©мһҗк°Җ лӢӨлҘё мқён„°лһҷм…ҳмқ„ кі„мҶҚн•  мҲҳ мһҲлҸ„лЎқ н•ҙмӨҖлӢӨ.мҳҲлҘј л“Өл©ҙ мң нҠңлёҢм—җм„ң лҸҷмҳҒмғҒ лЎңл”© мӨ‘м—җлҸ„ мӮ¬мҡ©мһҗк°Җ лҢ“кёҖмқ„ мқҪкұ°лӮҳ мўӢм•„мҡ”лҘј лҲ„лҘј мҲҳ мһҲлҠ” мқҙмң к°Җ л№„лҸҷкё° мІҳлҰ¬ лҚ•л¶„мқҙлӢӨ.

рҹ“қ нҳ„мӢӨм Ғмқё м• н”ҢлҰ¬мјҖмқҙм…ҳ мҡ”кө¬мӮ¬н•ӯ

нҳ„лҢҖ мӣ№ м• н”ҢлҰ¬мјҖмқҙм…ҳмқҖ лӢӨмӨ‘ мһ‘м—…, мҰү лҚ°мқҙн„° лЎңл“ң, мқҙлҜём§Җ л ҢлҚ”л§Ғ, мӮ¬мҡ©мһҗ мһ…л Ҙ мІҳлҰ¬мқ„ лҸҷмӢңм—җ мІҳлҰ¬н•ҙм•ј н•ҳлҠ”лҚ°, мқҙлҘј л№„лҸҷкё° л°©мӢқмңјлЎңл§Ң кө¬нҳ„н•  мҲҳ мһҲлӢӨ.

к°ҖлҸ…м„ұкіј мң м§ҖліҙмҲҳм„ұ к°ңм„  aync/await

async/awaitлҠ” ES2017(ECMAScript 2017)м—җм„ң лҸ„мһ…лҗң л№„лҸҷкё° мһ‘м—… мІҳлҰ¬ л¬ёлІ•мңјлЎң, л№„лҸҷкё° мҪ”л“ңлҘј ліҙлӢӨ к°„кІ°н•ҳкі  м§ҒкҙҖм ҒмңјлЎң мһ‘м„ұн•  мҲҳ мһҲкІҢ н•ҙмӨҖлӢӨ. кё°мЎҙмқҳ мҪңл°ұ н•ЁмҲҳлҘј м—°мҶҚм ҒмңјлЎң нҳём¶ңн•ҳлҠ” л°©мӢқмқҖ к°ҖлҸ…м„ұмқҙ л–Ём–ҙм§Җкі  лЎңм§Ғ мҲҳм •мқҙ м–ҙл Өмҡҙ "мҪңл°ұ м§ҖмҳҘ(callback hell)" л¬ём ңлҘј мқјмңјнӮ¬ мҲҳ мһҲлҠ”лҚ° async/awaitлҠ” мқҙлҹ¬н•ң л¬ём ңлҘј н•ҙкІ°н•ҳл©°, л№„лҸҷкё° мһ‘м—…мқ„ лҸҷкё° мҪ”л“ңмІҳлҹј мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.
  • async нӮӨмӣҢл“ңлЎң м •мқҳлҗң н•ЁмҲҳлҠ” н•ӯмғҒ Promise к°қмІҙлҘј л°ҳнҷҳн•ңлӢӨ.
  • async н•ЁмҲҳ лӮҙл¶Җм—җм„ң мӮ¬мҡ©лҗҳл©°, л№„лҸҷкё° мһ‘м—…(Promise)мқҙ мҷ„лЈҢлҗ  л•Ңк№Ңм§Җ кё°лӢӨлҰ° нӣ„ кІ°кіјлҘј л°ҳнҷҳн•ңлӢӨ.
// мҪңл°ұ л°©мӢқ
getData(function (data) {
  processData(data, function (result) {
    saveData(result, function (response) {
      console.log(response);
    });
  });
});

// async/await л°©мӢқ
async function handleData() {
  const data = await getData();
  const result = await processData(data);
  const response = await saveData(result);
  console.log(response);
}

PromiseлһҖ

PromiseлҠ” л№„лҸҷкё° мһ‘м—…мқ„ лӢӨлЈЁкё° мң„н•ң к°қмІҙлЎң, мһ‘м—…мқҳ м„ұкіө(resolve) лҳҗлҠ” мӢӨнҢЁ(reject) мғҒнғңлҘј лӘ…нҷ•нһҲ кҙҖлҰ¬н•  мҲҳ мһҲкі  л№„лҸҷкё° мһ‘м—…мқҳ нқҗлҰ„мқ„ мІҙкі„м ҒмңјлЎң мІҳлҰ¬н•ҳлҠ” лҚ° мң мҡ©н•©лӢҲлӢӨ.

рҹ“қ кө¬м„ұ

new Promise мғқм„ұмһҗлҘј нҶөн•ҙ м •мқҳлҗҳл©°, лӮҙл¶Җм—җм„ң resolve лҳҗлҠ” reject н•ЁмҲҳлҘј нҳём¶ңн•©лӢҲлӢӨ.
const myPromise = new Promise((resolve, reject) => {
  // л№„лҸҷкё° мһ‘м—… мӢңл®¬л Ҳмқҙм…ҳ (мҳҲ: 2мҙҲ нӣ„ кІ°кіј л°ҳнҷҳ)
  setTimeout(() => {
    const success = true; // м„ұкіө м—¬л¶ҖлҘј мӢңл®¬л Ҳмқҙм…ҳ
    if (success) {
      resolve("мһ‘м—…мқҙ м„ұкіөм ҒмңјлЎң мҷ„лЈҢлҗҳм—ҲмҠөлӢҲлӢӨ!"); // м„ұкіө мӢң resolve нҳём¶ң
    } else {
      reject("мһ‘м—…мқҙ мӢӨнҢЁн–ҲмҠөлӢҲлӢӨ."); // мӢӨнҢЁ мӢң reject нҳём¶ң
    }
  }, 2000);
});
